#include <algorithm>

const int N = 100010;
pair<long long, long long> a[N];
long long b[N];

int main() {
    int n;
    scanf("%d", &n);
    for (int i = 1; i <= n; ++i) scanf("%lld", &a[i].first);
    for (int i = 1; i <= n; ++i) scanf("%lld", &a[i].second);
    sort(a + 1, a + n + 1, [](pair<long long, long long> a, pair<long long, long long> b) {
        return a.first == b.first ? a.second >= b.second : a.first < b.first;
    });
    for (int i = n; i; --i) b[i] = max(b[i + 1], a[i].second);
    int res = 0;
    for (int i = 1; i <= n; ++i) {
        if (a[i].second == b[i]) res++;
    }
    printf("%d\n", res);
    return 0;
}

#include <iostream>
#include <algorithm>

using namespace std;

typedef long long LL;
const int N = 100010;

pair<LL, LL> a[N];

bool cmp(pair<LL, LL> a, pair<LL, LL> b) {
    return a.second < b.second;
}

int main() {
    int n;
    LL t, res = 0;
    scanf("%d%lld", &n, &t);
    for (int i = 1; i <= n; ++i) {
        scanf("%lld", &a[i].first);
    }
    for (int i = 1; i <= n; ++i) {
        scanf("%lld", &a[i].second);
    }
    sort(a + 1, a + n + 1, cmp);
    for (int i = 1; i <= n; ++i) {
        if (t >= a[i].first * a[i].second) {
            res += a[i].first;
            t -= a[i].first * a[i].second;
        }
        else {
            res += t / a[i].second;
            break;
        }
    }
    printf("%lld\n", res);
    return 0;
}
