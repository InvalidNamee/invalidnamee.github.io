const int MOD = 1000000007;

int main() {
    long long res = 0;
    int n;
    scanf("%d", &n);
    for (int i = 1; i <= n; ++i) {
        int t;
        scanf("%d", &t);
        res += (long long)i * (n - i + 1) % MOD * t % MOD;

void bfs(int i) {
    queue<int> q;
    q.push(i);
    vis[i] = tot;
    while (!q.empty()) {
        int x = q.front();
        q.pop();
        for (int y : ed[x]) {
            if (vis[y]) continue;
            vis[y] = tot;
            q.push(y);
        }
        if (!vis[pre[x]]) {
            vis[pre[x]] = tot;
            q.push(pre[x]);
        }
    }
}
