#include <iostream>
#include <cmath>

using namespace std;

bool check(long long t) {
    string s = to_string(t);
    for (int i = 0; i < s.length(); ++i) {
        if (s[i] != s[s.length() - i - 1]) return false;
    }
    return true;
}

int main() {
    long long n, t = 0;
    cin >> n;
    for (int i = 1; (long long)i * i * i <= n; ++i) {
        if (check((long long)i * i * i)) {
            t = (long long)i * i * i;
        }
    }
    printf("%lld\n", t);
    return 0;
}

#include <iostream>
#include <vector>
#include <queue>

using namespace std;

const int N = 100010;

int head[N], ne[N], ver[N], tot;
int w[N];
vector<long long> f[N];
int cnt[N];
int n, k;

void add(int x, int y) {
    ver[++tot] = y;
    ne[tot] = head[x];
    head[x] = tot;
}

void dp(int x) {
    cnt[x] = 1;
    long long s = 0;
    for (int i = head[x]; i; i = ne[i]) {
        int y = ver[i];
        dp(y);
        cnt[x] += cnt[y];
        for (int j = min(k + 1, cnt[x]); j; --j) {
            for (int l = min(cnt[y], j); l >= 0; --l) {
                f[x][j] = max(f[x][j], f[x][j - l] + f[y][l]);
            }
        }
    }
    f[x][1] = max(f[x][1], (long long)w[x]);
}

int main() {
    scanf("%d%d", &n, &k);
    for (int i = 2; i <= n; ++i) {
        int f;
        scanf("%d%d", &f, &w[i]);
        add(f, i);
    }
    for (int i = 1; i <= n; ++i) {
        f[i] = vector<long long>(k + 2, 0);
    }
    dp(1);
    long long res = 0;
    for (int i = 0; i <= k + 1; ++i) {
        res = max(res, f[1][i]);
    }
    printf("%lld\n", res);
    return 0;
}
