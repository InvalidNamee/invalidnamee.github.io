    cout << endl;

                D[k][i][j] = min(D[k - 1][i][j], D[k - 1][i][k] + D[k - 1][k][j]);

                else cout << setw(4) << D[k][i][j];
            }
            cout << endl;
        }
        cout << endl;
    }
    return 0;
}
