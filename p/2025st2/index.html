const int N = 200010;
vector<int> ed[N];
int id[N], tot;
int cnt[N], ed_cnt[N];

    if (id[x]) return; 
    id[x] = tot;
    cnt[tot]++;
    for (int y : ed[x]) {
        dfs(y);
    }
}

int main() {
    int n, m;

    for (int i = 0; i < m; ++i) {
        int x, y;
        scanf("%d%d", &x, &y);
        ed[x].push_back(y);
        ed[y].push_back(x);
    }
    for (int i = 1; i <= n; ++i) {
        sort(ed[i].begin(), ed[i].end());
        ed[i].erase(unique(ed[i].begin(), ed[i].end()), ed[i].end());
    }
    for (int i = 1; i <= n; ++i) {
        if (!id[i]) {
            ++tot;
            dfs(i);
        }
    }
    for (int i = 1; i <= n; ++i) {
        ed_cnt[id[i]] += ed[i].size();
    }
    long long res = 0;

        res += (long long)cnt[i] * (cnt[i] - 1) / 2 - ed_cnt[i] / 2;
    }
    cout << res << endl;

#include <iostream>
#include <cmath>
#include <unordered_set>

using namespace std;

long long power(long long n, long long p) {
    long long res = 1, base = n;
    while (p) {
        if (p & 1) res = res * base;
        base = base * base;
        p >>= 1;
    }
    return res;
}

unordered_set<long long> s;

int main() {
    long long n;
    cin >> n;
    for (int i = 60; i >= 3; --i) {
        for (int j = (long long)pow(n, 1.0 / i); j; --j) {
            long long t = power(j, i);
            if (power((int)sqrt(t), 2) != t) s.insert(t);
        }
    }
    cout << s.size() + (int)(sqrt(n) + 1e-9) << endl;
    return 0;
}

#include <iostream>
#include <vector>

using namespace std;

int cur[20];
int n;

bool dfs(int x, int d) {
    if (x > d) {
        if (cur[x - 1] == n) return true;
        else return false;
    }
    if (cur[x - 1] > n || (long long)cur[x - 1] * (1 << (d - x + 1)) < n) return false;
    for (int i = 0; i < x; ++i) {
        for (int j = 0; j <= i; ++j) {
            if (cur[i] + cur[j] > cur[x - 1]) {
                cur[x] = cur[i] + cur[j];
                if (dfs(x + 1, d)) return true;
            }
        }
    }
    return false;
}

int main() {
    scanf("%d", &n);
    int l;
    cur[0] = 1;
    for (l = 1; !dfs(1, l); ++l);
    for (int i = 1; i <= l; ++i) {
        printf("%d ", cur[i]);
    }
    printf("\n");
    return 0;
}
