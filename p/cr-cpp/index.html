def convert_formulation_to_text(formulation: Tag, output_path: Path):
    md_lines = []

    # ---- Ê≠£ÊñáÊñáÊú¨ÔºàË∑≥ËøáË°®Ê†ºÔºâ ----
    for tag in formulation.find_all(['h1', 'h2', 'h3', 'h4', 'h5', 'h6', 'p', 'div'], recursive=True):
        # Ë∑≥ËøáÂåÖÂê´Ë°®Ê†ºÁöÑ div/p
        if tag.find('table'):
            continue
        text = tag.get_text(strip=True, separator=' ')
        if text:
            md_lines.append(text)

    # ---- Ë°®Ê†ºÊ†∑‰æãÂ§ÑÁêÜ ----
    table = formulation.find('table')
    if table:
        for row in table.find_all('tr')[1:]:  # Ë∑≥ËøáË°®Â§¥
            cells = row.find_all('td')
            if len(cells) >= 2:
                input_text = cells[0].get_text(strip=True)
                output_text = cells[1].get_text(strip=True)
                md_lines.append(f"\nËæìÂÖ•Ôºö\n{input_text}\n\nËæìÂá∫Ôºö\n{output_text}")

    # ---- ‰ª£Á†ÅÂùó ----
    textarea = formulation.find('textarea', class_='coderunner-answer')
    if textarea:
        code = (
            textarea.text
            .replace('&lt;', '<')
            .replace('&gt;', '>')
            .replace('&amp;', '&')
            .strip()
        )
        md_lines.append(code)

    # ---- ÂÜôÂÖ•Êñá‰ª∂ ----
    output_path.write_text('\n'.join(md_lines), encoding='utf-8')
