#include <iostream>

using namespace std;

int f(int n, int m) {
    if (n < m) swap(n, m);
    int res = 0;
    for (int i = 1; i <= n; ++i) {
        int t = (i - 1) * m / n * n;
        if ((i * m - t) + ((i - 1) * m - t) == n) res++;
    }
    return res;
}

int main() {
    cout << f(22, 38) << endl << f(21, 36) << endl;
    return 0;
}

const int MOD = 1000000007;

vector<vector<int>> mul(vector<vector<int>> a, vector<vector<int>> b) {
    vector<vector<int>> res(2, vector<int>(2, 0));
    for (int k = 0; k < 2; ++k) {
        for (int i = 0; i < 2; ++i) {
            for (int j = 0; j < 2; ++j) {
                res[i][j] = (res[i][j] + (long long)a[i][k] * b[k][j] % MOD) % MOD;
            }
        }
    }
    return res;
}

int solve(unsigned long long p) {
    if (p == 3) return 25;
    vector<vector<int>> res = {{1, 0}, {0, 1}}, base = {{1, 2}, {1, 0}};
    while (p) {
        if (p & 1) res = mul(res, base);
        base = mul(base, base);
        p >>= 1;
    }
    return (res[0][0] + res[0][1]) % MOD;
}

    }
    int res = 0;
    for (int i = 1; i <= n; ++i) {
        res += s[i];
    }
    printf("%d\n", res);
    return 0;
}
