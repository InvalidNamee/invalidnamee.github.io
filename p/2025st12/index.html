#include <iostream>
#include <unordered_map>

using namespace std;

unordered_map<int, int> s;
int n;
struct FenwickTree
{
    int val[100001];
    bool mark[100001];
    void add(int u) {
        if (mark[u]) return;
        mark[u] = true;
        for (; u <= n; u += u & -u) {
            val[u] += 1;
        }
    }
    int query(int u) {
        int res = 0;
        for (; u; u -= u & -u) {
            res += val[u];
        }
        return res;
    }
} tr[101];

int main() {
    ios::sync_with_stdio(0);
    cin.tie(0), cout.tie(0);
    int q;
    cin >> q;
    n = q;
    while (q--) {
        char c;
        int x;
        cin >> c >> x;
        if (c == '+') {
            if (s[x] == 0) {
                for (int i = 1; i <= 100; ++i) {
                    if (x % i == 0) {
                        if (x / i <= 100000) tr[i].add(x / i);
                    }
                }
            }
            s[x]++;
        }
        else {
            if (x > 100) {
                int i;
                for (i = 1; s[i * x]; ++i);
                cout << i * x << endl;
            }
            else {
                int l = 1, r = 100000;
                while (l < r) {
                    int mid = l + r >> 1;
                    if (tr[x].query(mid) < mid) r = mid;
                    else l = mid + 1;
                }
                cout << l * x << endl;
            }
        }
    }
    return 0;
}
