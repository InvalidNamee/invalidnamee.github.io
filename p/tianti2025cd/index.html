    int n, m;

    for (int i = 1; i <= n; ++i) fa[i] = i;

            ed[x].push_back({w, y});

        else {
            int x, w;

                a[x] = w;
                f[getfa(x)] = true;

        }

            if (a[y]) continue;
            a[y] = a[x] ^ v;
            q.push(y);
        }

        }

}
